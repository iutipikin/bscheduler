93  decode(It first, It last) {
94  // read first two bytes of a frame
95  size_t input_size = last - first;
96  if (input_size < constants::BASE_SIZE)
97  return std::make_pair(first,last);
98  std::copy(first, first + constants::BASE_SIZE, rawhdr);
99  // keep reading until the end of the header
100  size_t hdrsz = header_size();
101  if (input_size < hdrsz) return std::make_pair(first,last);
102  if (hdrsz > constants::BASE_SIZE) {
103  std::copy(
104  first + constants::BASE_SIZE,
105  first + hdrsz,
106  rawhdr + constants::BASE_SIZE
107  );
108  }
109  size_t payload_sz = payload_size();
110  if (input_size < payload_sz + hdrsz) {
111  return std::make_pair(first,last);
112  }
113  return std::make_pair(first + hdrsz, first + hdrsz + payload_sz);
114  }
