128 std::valarray<T> determine_nit_coefs(const std::valarray<T>& a,
129  size_t max_coefs,
130  const std::valarray<T>& acf)
131 {
132  std::valarray<T> c(max_coefs);
133  T sum_c = 0;
134  T f = 1;
135  T err = std::numeric_limits<T>::max();
136  size_t trim = 0;
137  for (size_t m=0; m<c.size(); ++m) {
138 
139  // calc coef
140  Poly<T> y = Poly<T>(a) * hermite_poly<T>(m);
141  T sum2 = y[0];
142  for (size_t i=2; i<=y.order(); i+=2)
143  sum2 += y[i]*fact<T>(i-1, 2);
144  c[m] = sum2;
145 
146  // calc error
147  sum_c += c[m]*c[m]/f;
148  f *= (m+1);
149  T e = std::abs(acf[0] - sum_c);
150  // критерий возможно правильно писать так: abs(T(1) - sum_c)
151 
152  // determine minimum error
153  if (e < err) {
154  err = e;
155  trim = m+1;
156  }
157  std::clog << "err = " << e << std::endl;
158  }
159  std::clog << "trim = " << trim << std::endl;
160  return c[std::slice(0, trim, 1)];
161 }
