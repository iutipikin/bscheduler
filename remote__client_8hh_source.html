228  void
229  receive_kernel(kernel_type* k) {
230  bool ok = true;
231  k->from(_vaddr);
232  if (k->moves_downstream()) {
233  this->clear_kernel_buffer(k);
234  } else if (k->principal_id()) {
235  instances_guard g(instances);
236  auto result = instances.find(k->principal_id());
237  if (result == instances.end()) {
238  k->result(exit_code::no_principal_found);
239  ok = false;
240  }
241  k->principal(result->second);
242  }
243  #ifndef NDEBUG
244  sys::log_message("nic", "recv _", *k);
245  #endif
246  if (!ok) {
247  return_kernel(k);
248  } else {

265  #endif

272  k->result(exit_code::endpoint_not_connected);
273  k->principal(k->parent());
274  _router.send_local(k);
275  } else if (k->moves_downstream() and k->carries_parent()) {
276  #ifndef NDEBUG

278  #endif
279  _router.send_local(k);
