466  void
467  process_kernel(kernel_type* k) {
468  if (this->server_addr() && k->to() == this->server_addr()) {
469  _router.send_local(k);
470 // std::ostringstream msg;
471 // msg << "Kernel is sent to local node. From="
472 // << this->server_addr() << ", to=" << k->to();
473 // throw Error(msg.str(), __FILE__, __LINE__, __func__);
474  }
475 
476  if (k->moves_everywhere()) {
477  for (auto& pair : _upstream) {
478  pair.second->send(k);
479  }
480  // delete broadcast kernel
481  delete k;
482  } else if (k->moves_upstream() && k->to() == sys::endpoint()) {
483  if (_upstream.empty() or _endreached) {
484  // include localhost in round-robin
485  // in a somewhat half-assed fashion
486  _endreached = false;
487  // short-circuit kernels when no upstream servers are available
488  _router.send_local(k);
489  } else {
490  // skip stopped hosts
491  iterator_type old_iterator = _iterator;
492  if (_iterator->second->is_stopped()) {
493  do {
494  advance_upstream_iterator();
495  } while (_iterator->second->is_stopped() and old_iterator != _iterator);
496  }
497  // round robin over upstream hosts
498  ensure_identity(k);
499  _iterator->second->send(k);
500  advance_upstream_iterator();
501  }
502  } else if (k->moves_downstream() and not k->from()) {
503  // kernel @k was sent to local node
504  // because no upstream servers had
505  // been available
506  _router.send_local(k);
507  } else {
508  // create endpoint if necessary, and send kernel
509  if (not k->to()) {
510  k->to(k->from());
511  }
512  if (k->moves_somewhere()) {
513  ensure_identity(k);
514  }
515  this->find_or_create_peer(k->to(), sys::poll_event::Inout)->send(k);
516  }
517  }
